from django.test import TestCase, RequestFactory
from unittest.mock import MagicMock, patch
from django.contrib.admin.sites import AdminSite
from core.forms import ConfigAdminForm
from core.admin import ConfigAdmin
from core.models.config import Config


class ConfigAdminFormTest(TestCase):

    # Covers: new config string path
    def test_new_string_valid(self):
        form = ConfigAdminForm(data={
            'key': 'test_key', 'value_type': 'string',
            'value': 'hello', 'boolean_value': '', 'remarks': ''
        })
        self.assertTrue(form.is_valid())
        self.assertEqual(form.cleaned_data['value'], 'hello')

    # Covers: new config string empty value error
    def test_new_string_empty_value_invalid(self):
        form = ConfigAdminForm(data={
            'key': 'test_key', 'value_type': 'string',
            'value': '', 'boolean_value': '', 'remarks': ''
        })
        self.assertFalse(form.is_valid())

    # Covers: new config boolean path + cleaned_data['value'] = cleaned_data['boolean_value']
    def test_new_boolean_valid(self):
        form = ConfigAdminForm(data={
            'key': 'test_key', 'value_type': 'boolean',
            'value': '', 'boolean_value': 'true', 'remarks': ''
        })
        self.assertTrue(form.is_valid())
        self.assertEqual(form.cleaned_data['value'], 'true')

    # Covers: new config boolean with None selected error
    def test_new_boolean_none_selected_invalid(self):
        form = ConfigAdminForm(data={
            'key': 'test_key', 'value_type': 'boolean',
            'value': '', 'boolean_value': '', 'remarks': ''
        })
        self.assertFalse(form.is_valid())

    # Covers: existing boolean config __init__ initial setting + _configure_field hide value
    def test_existing_boolean_initial_preselected(self):
        instance = Config(pk=1, key='flag', value='true')
        form = ConfigAdminForm(instance=instance)
        self.assertEqual(form.initial['boolean_value'], 'true')

    # Covers: existing boolean config clean path
    def test_existing_boolean_update(self):
        instance = Config(pk=1, key='flag', value='true')
        form = ConfigAdminForm(
            data={'key': 'flag', 'boolean_value': 'false', 'remarks': ''},
            instance=instance
        )
        self.assertTrue(form.is_valid())
        self.assertEqual(form.cleaned_data['value'], 'false')

    # Covers: existing boolean none selected error
    def test_existing_boolean_none_invalid(self):
        instance = Config(pk=1, key='flag', value='true')
        form = ConfigAdminForm(
            data={'key': 'flag', 'boolean_value': '', 'remarks': ''},
            instance=instance
        )
        self.assertFalse(form.is_valid())

    # Covers: existing string config else branch in __init__
    def test_existing_string_update(self):
        instance = Config(pk=1, key='my_key', value='old')
        form = ConfigAdminForm(
            data={'key': 'my_key', 'value': 'new', 'remarks': ''},
            instance=instance
        )
        self.assertTrue(form.is_valid())


class ConfigAdminTest(TestCase):

    def setUp(self):
        self.site = AdminSite()
        self.admin = ConfigAdmin(Config, self.site)
        self.request = RequestFactory().get('/')

    # Covers: get_fieldsets new config
    def test_get_fieldsets_new_config(self):
        fieldsets = self.admin.get_fieldsets(self.request, obj=None)
        fields = fieldsets[0][1]['fields']
        self.assertIn('value_type', fields)
        self.assertIn('value', fields)
        self.assertIn('boolean_value', fields)

    # Covers: get_fieldsets existing boolean — field = 'boolean_value'
    def test_get_fieldsets_existing_boolean(self):
        obj = Config(pk=1, key='flag', value='true')
        fieldsets = self.admin.get_fieldsets(self.request, obj=obj)
        fields = fieldsets[0][1]['fields']
        self.assertIn('boolean_value', fields)
        self.assertNotIn('value', fields)

    # Covers: get_fieldsets existing string — field = 'value'
    def test_get_fieldsets_existing_string(self):
        obj = Config(pk=1, key='my_key', value='hello')
        fieldsets = self.admin.get_fieldsets(self.request, obj=obj)
        fields = fieldsets[0][1]['fields']
        self.assertIn('value', fields)
        self.assertNotIn('boolean_value', fields)

    # Covers: save_model — obj.value = form.cleaned_data['value'] + ConfigService.save
    @patch('core.services.config.ConfigService.save')
    def test_save_model(self, mock_save):
        obj = Config(pk=1, key='flag', value='')
        form = MagicMock()
        form.cleaned_data = {'value': 'true'}
        self.admin.save_model(self.request, obj, form, change=True)
        self.assertEqual(obj.value, 'true')
        mock_save.assert_called_once_with(obj, self.request.user)
