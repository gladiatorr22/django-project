from django import forms
from .models import Config

class ConfigAdminForm(forms.ModelForm):
    VALUE_TYPE_CHOICES = [
        ('string', 'String'),
        ('boolean', 'Boolean'),
    ]

    value_type = forms.ChoiceField(
        choices=VALUE_TYPE_CHOICES,
        widget=forms.Select,
        required=False,
        label='Value Type'
    )

    value = forms.ChoiceField(
        choices=[('true', 'True'), ('false', 'False')],
        widget=forms.RadioSelect,
        required=False,
        label='Value'
    )

    class Meta:
        model = Config
        fields = ['key', 'value_type', 'value', 'remarks']

def changeform_view(self, request, object_id=None, form_url='', extra_context=None):
        if request.method == 'POST':
            post_data = request.POST.copy()
            value_type = post_data.get('value_type')
            # If value_type is boolean but value is not true/false, just re-render
            if value_type == 'boolean' and post_data.get('value') not in ('true', 'false'):
                post_data['value'] = 'true'  # default
                request.POST = post_data
        return super().changeform_view(request, object_id, form_url, extra_context)
