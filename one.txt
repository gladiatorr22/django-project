class ConfigAdminForm(forms.ModelForm):
    VALUE_TYPE_CHOICES = [('string', 'String'), ('boolean', 'Boolean')]
    
    value_type = forms.ChoiceField(choices=VALUE_TYPE_CHOICES, required=False, label='Value Type')
    boolean_value = forms.ChoiceField(
        choices=[('true', 'True'), ('false', 'False')],
        widget=forms.RadioSelect,
        required=False,
        label='Boolean Value'
    )

    class Meta:
        model = Config
        fields = ['key', 'value_type', 'value', 'boolean_value', 'remarks']

    def __init__(self, *args, **kwargs):
        self.is_new = kwargs.pop('is_new', False)
        super().__init__(*args, **kwargs)
        self.fields['value'].required = False
        if not self.is_new:
            self.fields.pop('value_type', None)
            self.fields.pop('boolean_value', None)

    def clean(self):
        cleaned_data = super().clean()
        if not self.is_new:
            return cleaned_data
        
        value_type = cleaned_data.get('value_type')
        value = cleaned_data.get('value', '').strip()
        boolean_value = cleaned_data.get('boolean_value')

        if value_type == 'boolean':
            if not boolean_value:
                raise forms.ValidationError("Please select a boolean value.")
            cleaned_data['value'] = boolean_value  # saves 'true'/'false' into value field
        else:
            if not value:
                raise forms.ValidationError("Value is required.")
        
        return cleaned_data

class ConfigAdmin(DetailedLogModelAdmin):
    form = ConfigAdminForm
    list_per_page = 50
    search_fields = ("=key", "=updated_by__email", "=updated_by__username")
    readonly_fields = ("created_at", "modified_at", "updated_by")
    list_display = ("key", "value", "remarks", "updated_by", "modified_at")

    def get_fieldsets(self, request, obj=None):
        if obj is None:
            return [(None, {'fields': ('key', 'value_type', 'value', 'boolean_value', 'remarks')})]
        return [(None, {'fields': ('key', 'value', 'remarks')})]

    def get_form(self, request, obj=None, **kwargs):
        form_class = super().get_form(request, obj, **kwargs)
        is_new = obj is None
        class WrappedForm(form_class):
            def __init__(self, *args, **kw):
                kw['is_new'] = is_new
                super().__init__(*args, **kw)
        return WrappedForm

    def save_model(self, request, obj, form, change):
        if not change and form.cleaned_data.get('value_type') == 'boolean':
            obj.value = form.cleaned_data.get('boolean_value')
        ConfigService.save(obj, request.user)
