class ConfigAdminForm(forms.ModelForm):
    VALUE_TYPE_CHOICES = [('boolean', 'Boolean'), ('string', 'String')]

    value_type = forms.ChoiceField(
        choices=VALUE_TYPE_CHOICES,
        required=False,
        label='Value Type'
    )
    boolean_value = forms.ChoiceField(
        choices=[('', 'None'), ('true', 'True'), ('false', 'False')],
        widget=forms.RadioSelect,
        required=False,
        label='Boolean Value'
    )

    class Meta:
        model = Config
        fields = ['key', 'value_type', 'value', 'boolean_value', 'remarks']

    def __init__(self, *args, **kwargs):
        super().__init__(*args, **kwargs)
        self.fields['value'].required = False
        self.is_boolean = (
            self.instance and self.instance.value in ('true', 'false')
        )
        if self.instance.pk:
            # Existing config
            self.fields.pop('value_type')
            if self.is_boolean:
                self.fields['boolean_value'].initial = self.instance.value
                self.fields.pop('value')
            else:
                self.fields.pop('boolean_value')

    def clean(self):
        cleaned_data = super().clean()
        
        if self.instance.pk:
            # Existing config
            if self.is_boolean:
                if not cleaned_data.get('boolean_value'):
                    raise forms.ValidationError("Please select a boolean value.")
                cleaned_data['value'] = cleaned_data['boolean_value']
            return cleaned_data

        # New config
        value_type = cleaned_data.get('value_type')
        if value_type == 'boolean':
            if not cleaned_data.get('boolean_value'):
                raise forms.ValidationError("Please select a boolean value.")
            cleaned_data['value'] = cleaned_data['boolean_value']
        else:
            if not cleaned_data.get('value', '').strip():
                raise forms.ValidationError("Value is required.")

        return cleaned_data

class ConfigAdmin(DetailedLogModelAdmin):
    form = ConfigAdminForm
    list_per_page = 50
    search_fields = ("=key", "=updated_by__email", "=updated_by__username")
    readonly_fields = ("created_at", "modified_at", "updated_by")
    list_display = ("key", "value", "remarks", "updated_by", "modified_at")

    def get_fieldsets(self, request, obj=None):
        if obj is None:
            # New config - show all fields
            return [(None, {'fields': ('key', 'value_type', 'value', 'boolean_value', 'remarks')})]
        # Existing config - show based on value type
        if obj.value in ('true', 'false'):
            return [(None, {'fields': ('key', 'boolean_value', 'remarks')})]
        return [(None, {'fields': ('key', 'value', 'remarks')})]

    def save_model(self, request, obj, form, change):
        obj.value = form.cleaned_data.get('value')
        ConfigService.save(obj, request.user)




def __init__(self, *args, **kwargs):
    instance = kwargs.get('instance')
    if instance and instance.pk and instance.value in ('true', 'false'):
        initial = kwargs.get('initial', {})
        initial['boolean_value'] = instance.value  # set BEFORE super()
        kwargs['initial'] = initial
    
    super().__init__(*args, **kwargs)
    
    self.is_boolean = instance and instance.pk and instance.value in ('true', 'false')

    if instance and instance.pk:
        self.fields['value_type'].widget = forms.HiddenInput()
        if self.is_boolean:
            self.fields['value'].widget = forms.HiddenInput()
            self.fields['value'].required = False
            # Force override — this is the key line
            self.initial['boolean_value'] = instance.value
            self.fields['boolean_value'].initial = instance.value
        else:
            self.fields['boolean_value'].widget = forms.HiddenInput()
            self.fields['value'].required = True
    else:
        self.fields['value'].required = False
        self.fields['boolean_value'].required = False
